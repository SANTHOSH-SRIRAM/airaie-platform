# Plan 01-02: Top Nav, Sidebar Router, Right Panel, Vertical Toolbar

<objective>
Build the five remaining shell components that fill the AppShell layout zones created in Plan 01-01: a v2 top navigation bar with page tabs, a SidebarContentRouter that switches sidebar content based on uiStore.sidebarContentType, a resizable/collapsible RightPanel, a VerticalToolbar on the right edge, and a BottomBar for page-specific actions. Wire all components to react to route changes so the sidebar content type updates automatically when the user navigates.
</objective>

<must_haves>
1. Header renders: AIRAIE logo (left), page tabs (Dashboard, Workflows, Agents, Tools), search bar with Cmd+K, notification bell, user avatar
2. Active page tab has a visible underline indicator matching current route
3. SidebarContentRouter renders different sidebar content based on `uiStore.sidebarContentType`
4. When sidebarContentType is 'navigation', the existing nav menu (SidebarItem tree) renders
5. RightPanel component renders inside the right-panel-zone, reads width from `uiStore.rightPanel.width`, and has a drag handle for resizing
6. VerticalToolbar renders a vertical icon strip with at least: inspector toggle, search, settings icons
7. Clicking the inspector icon in VerticalToolbar calls `toggleRightPanel()`
8. BottomBar component renders inside the bottom-bar-zone with a slot for page-specific actions
9. Route changes automatically set `sidebarContentType` (e.g., '/workflows' -> 'navigation', '/workflow-studio' -> 'nodePalette')
10. All new components use CSS hiding (not conditional rendering) for toggling visibility
</must_haves>

import { useCallback, useRef } from 'react';
import { X } from 'lucide-react';
import { useUiStore } from '@store/uiStore';

const MIN_WIDTH = 260;
const MAX_WIDTH = 600;

export default function RightPanel() {
  const rightPanel = useUiStore((s) => s.rightPanel);
  const closeRightPanel = useUiStore((s) => s.closeRightPanel);
  const setRightPanelWidth = useUiStore((s) => s.setRightPanelWidth);
  const dragRef = useRef<{ startX: number; startWidth: number } | null>(null);

  const handleMouseDown = useCallback(
    (e: React.MouseEvent) => {
      e.preventDefault();
      dragRef.current = { startX: e.clientX, startWidth: rightPanel.width };

      const handleMouseMove = (moveEvent: MouseEvent) => {
        if (!dragRef.current) return;
        // Dragging left = making wider (right panel is on the right side)
        const delta = dragRef.current.startX - moveEvent.clientX;
        const newWidth = Math.max(MIN_WIDTH, Math.min(MAX_WIDTH, dragRef.current.startWidth + delta));
        setRightPanelWidth(newWidth);
      };

      const handleMouseUp = () => {
        dragRef.current = null;
        document.removeEventListener('mousemove', handleMouseMove);
        document.removeEventListener('mouseup', handleMouseUp);
        document.body.style.cursor = '';
        document.body.style.userSelect = '';
      };

      document.addEventListener('mousemove', handleMouseMove);
      document.addEventListener('mouseup', handleMouseUp);
      document.body.style.cursor = 'col-resize';
      document.body.style.userSelect = 'none';
    },
    [rightPanel.width, setRightPanelWidth]
  );

  return (
    <div className="h-full flex" data-testid="right-panel">
      {/* Drag handle (left edge) */}
      <div
        className="w-1 cursor-col-resize hover:bg-blue-50/50 active:bg-blue-50 transition-colors shrink-0"
        onMouseDown={handleMouseDown}
        role="separator"
        aria-orientation="vertical"
        aria-label="Resize right panel"
        tabIndex={0}
      />

      {/* Panel content */}
      <div className="flex-1 flex flex-col min-w-0">
        {/* Panel header */}
        <div className="h-10 flex items-center justify-between px-3 border-b border-cds-border-subtle shrink-0">
          <span className="text-sm font-medium text-cds-text-primary truncate">
            {rightPanel.contentType ?? 'Inspector'}
          </span>
          <button
            onClick={closeRightPanel}
            className="w-6 h-6 flex items-center justify-center text-cds-icon-secondary hover:text-cds-text-primary hover:bg-cds-background-hover rounded-sm transition-colors"
            aria-label="Close panel"
          >
            <X size={14} />
          </button>
        </div>

        {/* Panel body — content injected per context */}
        <div className="flex-1 overflow-y-auto p-3">
          {/* Phase-specific content components will be rendered here based on rightPanel.contentType */}
          <div data-testid="right-panel-content" className="text-sm text-cds-text-secondary">
            {/* Placeholder — replaced in Phase 3+ */}
          </div>
        </div>
      </div>
    </div>
  );
}
```

import type { SidebarContentType } from '@store/uiStore';

/** Maps route prefixes to sidebar content types.
 *  First match wins — order matters (more specific prefixes first). */
const ROUTE_SIDEBAR_MAP: Array<{ prefix: string; contentType: SidebarContentType }> = [
  { prefix: '/workflow-studio', contentType: 'nodePalette' },
  { prefix: '/agent-studio', contentType: 'sessions' },
  { prefix: '/tools', contentType: 'filters' },
  { prefix: '/', contentType: 'navigation' }, // catch-all last
];
```

3. Add a `useEffect` after the existing fullscreen reset effect:

```tsx
  // Sync sidebar content type with current route
  useEffect(() => {
    const match = ROUTE_SIDEBAR_MAP.find((entry) => pathname.startsWith(entry.prefix));
    if (match) {
      setSidebarContentType(match.contentType);
    }
  }, [pathname, setSidebarContentType]);
```

This ensures:
- `/workflow-studio/*` -> sidebar shows node palette
- `/agent-studio/*` -> sidebar shows sessions list
- `/tools/*` -> sidebar shows filters
- Everything else -> sidebar shows navigation menu
