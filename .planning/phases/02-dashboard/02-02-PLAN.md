# Plan 02-02: Dashboard Widgets and System Status Bar

<objective>
Build the four dashboard content widgets (Active Runs, Recent Workflows, Agent Activity, Governance) and the System Status Bar. Compose them into the DashboardPage replacing the placeholder slots from Plan 02-01. Wire the bottom bar zone in AppShell to show SystemStatusBar when the user is on the dashboard route. After this plan, the full dashboard is functional with all six visual sections from the Figma design.
</objective>

<must_haves>
- ActiveRunsWidget renders a list of running workflows with name, run ID, node progress fraction, elapsed time, cost, progress bar, and "View All" link
- RecentWorkflowsWidget renders a list of workflows with name, version badge (published green / draft gray), node count, relative timestamp, and status dot

- GovernanceWidget renders studies with name, gate progress bar (N/M gates), approval status badge, and action buttons (Study/Explore/Review)
- SystemStatusBar renders in the bottom bar zone: overall status indicator, API latency, NATS connection, runner slots, storage usage
- DashboardPage composes all widgets in a 2-column grid layout matching the Figma design
- Bottom bar automatically shows SystemStatusBar when on `/dashboard` route and hides when navigating away
</must_haves>

```tsx
import { ArrowRight } from 'lucide-react';
import { Link } from 'react-router-dom';
import Card from '@components/ui/Card';
import ProgressBar from '@components/ui/ProgressBar';
import Badge from '@components/ui/Badge';
import { useActiveRuns } from '@hooks/useDashboard';

function formatElapsed(seconds: number): string {
  const m = Math.floor(seconds / 60);
  const s = seconds % 60;
  return m > 0 ? `${m}m ${s}s` : `${s}s`;
}

function statusVariant(status: string): 'success' | 'warning' | 'info' {
  if (status === 'running') return 'success';
  if (status === 'waiting') return 'warning';
  return 'info';
}

export default function ActiveRunsWidget() {
  const { data: runs, isLoading } = useActiveRuns();

  return (
    <Card>
      <Card.Header className="flex items-center justify-between">
        <h2 className="text-sm font-semibold text-cds-text-primary">Active Runs</h2>
        <Link to="/workflows" className="text-xs text-cds-text-secondary hover:text-cds-text-primary flex items-center gap-1">
          View All <ArrowRight size={12} />
        </Link>
      </Card.Header>
      <Card.Body className="!p-0">
        {isLoading ? (
          <div className="p-4 space-y-3">
            {[1, 2, 3].map((i) => (
              <div key={i} className="h-16 bg-gray-10 animate-pulse rounded" />
            ))}
          </div>
        ) : !runs || runs.length === 0 ? (
          <div className="p-6 text-center text-sm text-cds-text-helper">No active runs</div>
        ) : (
          <ul className="divide-y divide-card-border-inner">
            {runs.map((run) => {
              const pct = run.nodesTotal > 0 ? Math.round((run.nodesCompleted / run.nodesTotal) * 100) : 0;
              return (
                <li key={run.id} className="px-4 py-3 hover:bg-surface-hover transition-colors">
                  <div className="flex items-center justify-between mb-1.5">
                    <div className="flex items-center gap-2">
                      <span className="text-sm font-medium text-cds-text-primary">{run.name}</span>
                      <Badge variant={statusVariant(run.status)} badgeStyle="outline">
                        {run.status}
                      </Badge>
                    </div>
                    <span className="text-xs font-mono text-cds-text-helper">{run.runId}</span>
                  </div>
                  <div className="flex items-center justify-between mb-1.5">
                    <span className="text-xs text-cds-text-secondary">
                      {run.nodesCompleted}/{run.nodesTotal} nodes
                    </span>
                    <div className="flex items-center gap-3 text-xs text-cds-text-helper">
                      <span>{formatElapsed(run.elapsedSeconds)}</span>
                      <span>${run.costUsd.toFixed(2)}</span>
                    </div>
                  </div>
                  <ProgressBar
                    value={pct}
                    max={100}
                    variant={run.status === 'running' ? 'blue' : 'default'}
                    size="sm"
                    showPercent={false}
                    striped={run.status === 'running'}
                  />
                </li>
              );
            })}
          </ul>
        )}
      </Card.Body>
    </Card>
  );
}
```

```tsx
import Card from '@components/ui/Card';
import Badge from '@components/ui/Badge';
import ProgressBar from '@components/ui/ProgressBar';
import Button from '@components/ui/Button';
import { useGovernance } from '@hooks/useDashboard';
import type { GovernanceStudy } from '@types/index';

function approvalVariant(status: GovernanceStudy['approvalStatus']): 'success' | 'warning' | 'danger' | 'info' {
  switch (status) {
    case 'approved': return 'success';
    case 'pending': return 'warning';
    case 'rejected': return 'danger';
    case 'in-review': return 'info';
  }
}

function approvalLabel(status: GovernanceStudy['approvalStatus']): string {
  switch (status) {
    case 'approved': return 'Approved';
    case 'pending': return 'Pending';
    case 'rejected': return 'Rejected';
    case 'in-review': return 'In Review';
  }
}

function actionLabel(action: string): string {
  return action.charAt(0).toUpperCase() + action.slice(1);
}

export default function GovernanceWidget() {
  const { data: studies, isLoading } = useGovernance();

  return (
    <Card>
      <Card.Header>
        <h2 className="text-sm font-semibold text-cds-text-primary">Governance</h2>
      </Card.Header>
      <Card.Body className="!p-0">
        {isLoading ? (
          <div className="p-4 space-y-3">
            {[1, 2, 3].map((i) => (
              <div key={i} className="h-20 bg-gray-10 animate-pulse rounded" />
            ))}
          </div>
        ) : !studies || studies.length === 0 ? (
          <div className="p-6 text-center text-sm text-cds-text-helper">No governance studies</div>
        ) : (
          <ul className="divide-y divide-card-border-inner">
            {studies.map((study) => {
              const gatePct = study.gatesTotal > 0 ? Math.round((study.gatesPassed / study.gatesTotal) * 100) : 0;
              const gateVariant = gatePct === 100 ? 'green' : gatePct >= 50 ? 'blue' : 'amber';
              return (
                <li key={study.id} className="px-4 py-3">
                  <div className="flex items-center justify-between mb-2">
                    <span className="text-sm font-medium text-cds-text-primary">{study.name}</span>
                    <Badge variant={approvalVariant(study.approvalStatus)} badgeStyle="filled">
                      {approvalLabel(study.approvalStatus)}
                    </Badge>
                  </div>
                  <ProgressBar
                    value={study.gatesPassed}
                    max={study.gatesTotal}
                    variant={gateVariant}
                    size="sm"
                    label={`${study.gatesPassed}/${study.gatesTotal} gates`}
                    showPercent={false}
                  />
                  <div className="flex items-center gap-2 mt-2">
                    {study.actions.map((action) => (
                      <Button key={action} variant="ghost" size="sm" className="!h-7 !px-3 !text-xs">
                        {actionLabel(action)}
                      </Button>
                    ))}
                  </div>
                </li>
              );
            })}
          </ul>
        )}
      </Card.Body>
    </Card>
  );
}
```

- Gate progress bar: value=gatesPassed, max=gatesTotal, label shows "N/M gates"
  - Color: green if 100%, blue if >= 50%, amber otherwise
- Action buttons row: ghost variant, small size (h-7), text-xs, capitalized labels

```tsx
import { Circle, Wifi, WifiOff, HardDrive, Cpu } from 'lucide-react';
import { useSystemHealth } from '@hooks/useDashboard';
import { formatBytes } from '@utils/format';
import { cn } from '@utils/cn';

const statusColor: Record<string, string> = {
  operational: 'text-green-50',
  degraded: 'text-yellow-30',
  outage: 'text-red-50',
};

const statusLabel: Record<string, string> = {
  operational: 'System Operational',
  degraded: 'System Degraded',
  outage: 'System Outage',
};

export default function SystemStatusBar() {
  const { data: status } = useSystemHealth();

  if (!status) return null;

  return (
    <div className="h-full flex items-center justify-between px-4 text-xs text-cds-text-secondary">
      {/* Left: overall status */}
      <div className="flex items-center gap-2">
        <Circle size={8} className={cn('fill-current', statusColor[status.overall])} />
        <span className="font-medium">{statusLabel[status.overall]}</span>
      </div>

      {/* Right: metrics */}
      <div className="flex items-center gap-4">
        <span>API: {status.apiLatencyMs}ms</span>

        <span className="flex items-center gap-1">
          {status.natsConnected ? (
            <Wifi size={12} className="text-green-50" />
          ) : (
            <WifiOff size={12} className="text-red-50" />
          )}
          NATS {status.natsConnected ? 'connected' : 'disconnected'}
        </span>

        <span className="flex items-center gap-1">
          <Cpu size={12} />
          Runners: {status.runnerSlots.used}/{status.runnerSlots.total}
        </span>

        <span className="flex items-center gap-1">
          <HardDrive size={12} />
          Storage: {formatBytes(status.storageUsed.bytes)} / {formatBytes(status.storageUsed.totalBytes)}
        </span>
      </div>
    </div>
  );
}
```
