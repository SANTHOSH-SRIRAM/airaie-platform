# Plan 03-01: Dependencies, Types, Store, and API Layer

<objective>
Install ReactFlow, extend workflow types for the visual editor, create a dedicated Zustand store for editor state, build the API layer with mock fallback, create React Query hooks, and implement conversion utilities between API workflow format and ReactFlow nodes/edges. This wave delivers all data infrastructure so Wave 2 and Wave 3 can build UI components without any foundational work.
</objective>

<must_haves>
- @xyflow/react is installed and chunked separately in vite build
- WorkflowNodeData, WorkflowNodeType, and node category types exist
- workflowStore manages nodes, edges, selectedNodeId, workflow metadata, dirty flag
- API module has CRUD + run with mock data fallback
- React Query hooks wrap all API calls
- Conversion utils can round-trip API workflow <-> ReactFlow format
- Node category constants define all 6 categories with their node definitions
</must_haves>

```ts
import type { Node, Edge } from '@xyflow/react';
import type { NodeCategory } from '@constants/nodeCategories';

/** Discriminated node type string used in the editor */
export type WorkflowNodeType =
  // Triggers
  | 'trigger-webhook' | 'trigger-schedule' | 'trigger-event'

  // Logic

  | 'data-transform' | 'data-store';

  label: string;
  nodeType: WorkflowNodeType;
  category: NodeCategory;
  icon: string;
  description?: string;
  toolVersion?: string;
  inputParameters?: Record<string, unknown>;

    lastRunAt?: string;
    avgCostUsd?: number;
  };
  config?: Record<string, unknown>;
}

export type WorkflowEditorNode = Node<WorkflowNodeData>;

/** Typed ReactFlow edge for the workflow editor */
export type WorkflowEditorEdge = Edge;

/** Workflow version status */

  id: string;

  versionStatus: WorkflowVersionStatus;
  ownerId: string;
```

```ts
import { create } from 'zustand';
import {
  applyNodeChanges,
  applyEdgeChanges,
  addEdge,
  type OnNodesChange,
  type OnEdgesChange,
  type OnConnect,
} from '@xyflow/react';
import type {
  WorkflowEditorNode,
  WorkflowEditorEdge,
  WorkflowEditorMetadata,
  WorkflowVersionStatus,
} from '@/types/workflow';

export interface WorkflowStore {
  // --- Graph state ---
  nodes: WorkflowEditorNode[];
  edges: WorkflowEditorEdge[];

  // --- Selection ---
  selectedNodeId: string | null;

  // --- Workflow metadata ---
  metadata: WorkflowEditorMetadata | null;

  // --- Dirty tracking ---
  isDirty: boolean;

  // --- Editor mode ---
  editorTab: 'editor' | 'runs' | 'eval';

  // --- Graph actions ---
  onNodesChange: OnNodesChange<WorkflowEditorNode>;
  onEdgesChange: OnEdgesChange;
  onConnect: OnConnect;
  setNodes: (nodes: WorkflowEditorNode[]) => void;
  setEdges: (edges: WorkflowEditorEdge[]) => void;
  addNode: (node: WorkflowEditorNode) => void;
  removeNode: (nodeId: string) => void;
  updateNodeData: (nodeId: string, data: Partial<WorkflowEditorNode['data']>) => void;

  // --- Selection actions ---
  selectNode: (nodeId: string | null) => void;

  // --- Metadata actions ---
  setMetadata: (metadata: WorkflowEditorMetadata) => void;
  setWorkflowName: (name: string) => void;
  setVersionStatus: (status: WorkflowVersionStatus) => void;

  // --- Dirty tracking ---
  markDirty: () => void;
  markClean: () => void;

  // --- Tab ---
  setEditorTab: (tab: 'editor' | 'runs' | 'eval') => void;

  // --- Reset ---
  resetEditor: () => void;

  // --- Load workflow ---
  loadWorkflow: (
    metadata: WorkflowEditorMetadata,
    nodes: WorkflowEditorNode[],
    edges: WorkflowEditorEdge[]
  ) => void;
}

const initialState = {
  nodes: [] as WorkflowEditorNode[],
  edges: [] as WorkflowEditorEdge[],
  selectedNodeId: null as string | null,
  metadata: null as WorkflowEditorMetadata | null,
  isDirty: false,
  editorTab: 'editor' as const,
};

export const useWorkflowStore = create<WorkflowStore>((set, get) => ({
  ...initialState,

  // --- Graph actions ---
  onNodesChange: (changes) => {
    set({
      nodes: applyNodeChanges(changes, get().nodes),
      isDirty: true,
    });
  },

  onEdgesChange: (changes) => {
    set({
      edges: applyEdgeChanges(changes, get().edges),
      isDirty: true,
    });
  },

  onConnect: (connection) => {
    set({
      edges: addEdge(connection, get().edges),
      isDirty: true,
    });
  },

  setNodes: (nodes) => set({ nodes, isDirty: true }),

  setEdges: (edges) => set({ edges, isDirty: true }),

  addNode: (node) =>
    set((s) => ({ nodes: [...s.nodes, node], isDirty: true })),

  removeNode: (nodeId) =>
    set((s) => ({
      nodes: s.nodes.filter((n) => n.id !== nodeId),
      edges: s.edges.filter((e) => e.source !== nodeId && e.target !== nodeId),
      selectedNodeId: s.selectedNodeId === nodeId ? null : s.selectedNodeId,
      isDirty: true,
    })),

  updateNodeData: (nodeId, data) =>
    set((s) => ({
      nodes: s.nodes.map((n) =>
        n.id === nodeId ? { ...n, data: { ...n.data, ...data } } : n
      ),
      isDirty: true,
    })),

  // --- Selection ---
  selectNode: (nodeId) => set({ selectedNodeId: nodeId }),

  // --- Metadata ---
  setMetadata: (metadata) => set({ metadata }),

  setWorkflowName: (name) =>
    set((s) => ({
      metadata: s.metadata ? { ...s.metadata, name } : null,
      isDirty: true,
    })),

  setVersionStatus: (status) =>
    set((s) => ({
      metadata: s.metadata
        ? { ...s.metadata, versionStatus: status }
        : null,
    })),

  // --- Dirty ---
  markDirty: () => set({ isDirty: true }),
  markClean: () => set({ isDirty: false }),

  // --- Tab ---
  setEditorTab: (tab) => set({ editorTab: tab }),

  // --- Reset ---
  resetEditor: () => set({ ...initialState }),

  // --- Load ---
  loadWorkflow: (metadata, nodes, edges) =>
    set({
      metadata,
      nodes,
      edges,
      selectedNodeId: null,
      isDirty: false,
      editorTab: 'editor',
    }),
}));
```
</action>

```ts
import type {
  Workflow,
  WorkflowStep,
  WorkflowEditorNode,
  WorkflowEditorEdge,
  WorkflowNodeData,
  WorkflowNodeType,
} from '@/types/workflow';
import { NODE_DEFINITION_MAP } from '@constants/nodeCategories';

/**
 * Map a WorkflowStep.type to a ReactFlow custom node type string.
 * This determines which custom node component renders.
 */
function stepTypeToReactFlowType(nodeType: WorkflowNodeType): string {
  const category = NODE_DEFINITION_MAP[nodeType]?.category;
  switch (category) {
    case 'triggers':
      return 'triggerNode';
    case 'tools':
      return 'toolNode';
    case 'agents':
      return 'agentNode';
    case 'logic':
      return 'logicNode';
    case 'governance':
      return 'governanceNode';
    case 'data':
      return 'dataNode';
    default:
      return 'toolNode';
  }
}

/**
 * Convert API WorkflowSteps to ReactFlow nodes.
 */
export function stepsToNodes(steps: WorkflowStep[]): WorkflowEditorNode[] {
  return steps.map((step) => {
    const nodeType = (step.config?.nodeType as WorkflowNodeType) ?? 'tool-fea-solver';
    const definition = NODE_DEFINITION_MAP[nodeType];

    const data: WorkflowNodeData = {
      label: step.name,
      nodeType,
      category: definition?.category ?? 'tools',
      icon: definition?.icon ?? 'Box',
      description: definition?.description,
      config: step.config,
    };

    return {
      id: step.id,
      type: stepTypeToReactFlowType(nodeType),
      position: step.position,
      data,
    };
  });
}

/**
 * Convert API WorkflowStep connections to ReactFlow edges.
 */
export function stepsToEdges(steps: WorkflowStep[]): WorkflowEditorEdge[] {
  const edges: WorkflowEditorEdge[] = [];
  for (const step of steps) {
    for (const targetId of step.connections) {
      edges.push({
        id: `edge-${step.id}-${targetId}`,
        source: step.id,
        target: targetId,
        animated: true,
      });
    }
  }
  return edges;
}

/**
 * Convert ReactFlow nodes back to API WorkflowSteps.
 */
export function nodesToSteps(
  nodes: WorkflowEditorNode[],
  edges: WorkflowEditorEdge[]
): WorkflowStep[] {
  // Build adjacency map from edges
  const connectionMap = new Map<string, string[]>();
  for (const edge of edges) {
    const existing = connectionMap.get(edge.source) ?? [];
    existing.push(edge.target);
    connectionMap.set(edge.source, existing);
  }

  return nodes.map((node) => ({
    id: node.id,
    name: node.data.label,
    type: categoryToStepType(node.data.category),
    status: 'pending' as const,
    position: node.position,
    connections: connectionMap.get(node.id) ?? [],
    config: {
      ...node.data.config,
      nodeType: node.data.nodeType,
    },
  }));
}

/**
 * Map node category to WorkflowStep type.
 */
function categoryToStepType(
  category: string
): 'action' | 'condition' | 'loop' | 'parallel' {
  switch (category) {
    case 'logic':
      return 'condition';
    default:
      return 'action';
  }
}

/**
 * Generate a unique node ID.
 */
export function generateNodeId(): string {
  return `node-${Date.now()}-${Math.random().toString(36).slice(2, 7)}`;
}

/**
 * Create a new WorkflowEditorNode from a node type at a given position.
 */
export function createNodeFromType(
  nodeType: WorkflowNodeType,
  position: { x: number; y: number }
): WorkflowEditorNode {
  const definition = NODE_DEFINITION_MAP[nodeType];
  if (!definition) {
    throw new Error(`Unknown node type: ${nodeType}`);
  }

  return {
    id: generateNodeId(),
    type: stepTypeToReactFlowType(nodeType),
    position,
    data: {
      label: definition.label,
      nodeType: definition.type,
      category: definition.category,
      icon: definition.icon,
      description: definition.description,
      config: { ...definition.defaultConfig },
    },
  };
}
```
