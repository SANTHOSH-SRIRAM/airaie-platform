# Plan 03-02: Canvas, Node Components, and Node Palette

<objective>
Build the visual workflow editor UI: a ReactFlow canvas with custom node components for each category, a searchable node palette sidebar with drag-to-canvas support, and a new WorkflowEditorPage that replaces the iframe-based WorkflowStudioPage. After this wave, users can view a workflow graph, drag nodes from the palette onto the canvas, and connect them.
</objective>

<must_haves>
- ReactFlow canvas renders with custom node types and handles zoom/pan
- Custom node components render for triggers, tools, agents, and logic categories
- Node palette sidebar shows all 6 categories with search filtering
- Drag from palette to canvas creates a new node at the drop position
- WorkflowEditorPage is lazy-loaded via React.lazy
- Route /workflow-studio renders the native editor (not the iframe)
- SidebarContentRouter mounts NodePalette in the nodePalette slot
</must_haves>

```tsx
        position={Position.Right}
```

```tsx
import { useCallback, useMemo, type DragEvent } from 'react';
import {
  ReactFlow,
  Background,
  Controls,
  MiniMap,
  BackgroundVariant,
  type NodeTypes,
  type ReactFlowInstance,
} from '@xyflow/react';
import '@xyflow/react/dist/style.css';
import { useWorkflowStore } from '@store/workflowStore';
import { createNodeFromType } from '@utils/workflowGraph';
import type { WorkflowNodeType } from '@/types/workflow';
import TriggerNode from './nodes/TriggerNode';
import ToolNode from './nodes/ToolNode';
import AgentNode from './nodes/AgentNode';
import LogicNode from './nodes/LogicNode';

/** Register all custom node types */
const nodeTypes: NodeTypes = {
  triggerNode: TriggerNode,
  toolNode: ToolNode,
  agentNode: AgentNode,
  logicNode: LogicNode,
  // governance and data nodes reuse toolNode styling for now
  governanceNode: ToolNode,
  dataNode: ToolNode,
};

let reactFlowInstance: ReactFlowInstance | null = null;

export default function WorkflowCanvas() {
  const nodes = useWorkflowStore((s) => s.nodes);
  const edges = useWorkflowStore((s) => s.edges);
  const onNodesChange = useWorkflowStore((s) => s.onNodesChange);
  const onEdgesChange = useWorkflowStore((s) => s.onEdgesChange);
  const onConnect = useWorkflowStore((s) => s.onConnect);
  const addNode = useWorkflowStore((s) => s.addNode);
  const selectNode = useWorkflowStore((s) => s.selectNode);

  const handleNodeClick = useCallback(
    (_event: React.MouseEvent, node: { id: string }) => {
      selectNode(node.id);
    },
    [selectNode]
  );

  const handlePaneClick = useCallback(() => {
    selectNode(null);
  }, [selectNode]);

  const handleInit = useCallback((instance: ReactFlowInstance) => {
    reactFlowInstance = instance;
  }, []);

  /** Handle drop from NodePalette */
  const handleDragOver = useCallback((event: DragEvent) => {
    event.preventDefault();
    event.dataTransfer.dropEffect = 'move';
  }, []);

  const handleDrop = useCallback(
    (event: DragEvent) => {
      event.preventDefault();

      const nodeType = event.dataTransfer.getData('application/workflow-node-type') as WorkflowNodeType;
      if (!nodeType || !reactFlowInstance) return;

      const position = reactFlowInstance.screenToFlowPosition({
        x: event.clientX,
        y: event.clientY,
      });

      const newNode = createNodeFromType(nodeType, position);
      addNode(newNode);
    },
    [addNode]
  );

  const defaultEdgeOptions = useMemo(
    () => ({
      animated: true,
      style: { stroke: '#94a3b8', strokeWidth: 2 },
    }),
    []
  );

  return (
    <div className="w-full h-full" data-testid="workflow-canvas">
      <ReactFlow
        nodes={nodes}
        edges={edges}
        onNodesChange={onNodesChange}
        onEdgesChange={onEdgesChange}
        onConnect={onConnect}
        onNodeClick={handleNodeClick}
        onPaneClick={handlePaneClick}
        onInit={handleInit}
        onDragOver={handleDragOver}
        onDrop={handleDrop}
        nodeTypes={nodeTypes}
        defaultEdgeOptions={defaultEdgeOptions}
        fitView
        fitViewOptions={{ padding: 0.2 }}
        deleteKeyCode={['Backspace', 'Delete']}
        multiSelectionKeyCode="Shift"
        proOptions={{ hideAttribution: true }}
      >
        <Background variant={BackgroundVariant.Dots} gap={20} size={1} color="#e2e8f0" />
        <Controls
          position="bottom-right"
          className="!bg-white !border !border-gray-200 !shadow-sm !rounded-lg"
        />
        <MiniMap
          position="bottom-left"
          className="!bg-white !border !border-gray-200 !shadow-sm !rounded-lg"
          maskColor="rgba(0, 0, 0, 0.08)"
          nodeColor={(node) => {
            const category = (node.data as Record<string, unknown>)?.category as string;
            switch (category) {
              case 'triggers': return '#f59e0b';
              case 'tools': return '#3b82f6';
              case 'agents': return '#8b5cf6';
              case 'logic': return '#10b981';
              case 'governance': return '#ef4444';
              case 'data': return '#06b6d4';
              default: return '#94a3b8';
            }
          }}
        />
      </ReactFlow>
    </div>
  );
}
```

```tsx
import { useState, useCallback, type DragEvent } from 'react';
import { Search } from 'lucide-react';
import * as LucideIcons from 'lucide-react';
import { cn } from '@utils/cn';
import { NODE_CATEGORIES, type NodeDefinition } from '@constants/nodeCategories';
import { NODE_CATEGORY_STYLES } from './nodes/nodeStyles';

/** Resolve a lucide icon by name string */
function getIcon(name: string): React.ComponentType<{ size?: number; className?: string }> {
  const icons = LucideIcons as unknown as Record<string, React.ComponentType<{ size?: number; className?: string }>>;
  return icons[name] ?? LucideIcons.Box;
}

export default function NodePalette() {
  const [search, setSearch] = useState('');

  const filteredCategories = NODE_CATEGORIES.map((cat) => ({
    ...cat,
    nodes: cat.nodes.filter(
      (n) =>
        n.label.toLowerCase().includes(search.toLowerCase()) ||
        n.description.toLowerCase().includes(search.toLowerCase())
    ),
  })).filter((cat) => cat.nodes.length > 0);

  const handleDragStart = useCallback((event: DragEvent, node: NodeDefinition) => {
    event.dataTransfer.setData('application/workflow-node-type', node.type);
    event.dataTransfer.effectAllowed = 'move';
  }, []);

  return (
    <div className="flex flex-col h-full" data-testid="node-palette">
      {/* Header */}
      <div className="px-4 pt-3 pb-2">
        <h3 className="text-xs font-medium text-cds-text-secondary uppercase tracking-wider mb-3">
          Nodes
        </h3>
        <div className="relative">
          <Search size={14} className="absolute left-3 top-1/2 -translate-y-1/2 text-cds-icon-secondary" />
          <input
            type="text"
            placeholder="Search nodes..."
            value={search}
            onChange={(e) => setSearch(e.target.value)}
            className={cn(
              'w-full h-8 pl-8 pr-3 text-xs bg-cds-field-01 text-cds-text-primary',
              'placeholder:text-cds-text-placeholder',
              'border-0 border-b border-cds-border-strong rounded-none',
              'focus:outline-2 focus:outline-cds-focus focus:outline-offset-[-2px]'
            )}
            aria-label="Search nodes"
          />
        </div>
      </div>

      {/* Category groups */}
      <div className="flex-1 overflow-y-auto px-2 pb-4">
        {filteredCategories.map((category) => {
          const catStyles = NODE_CATEGORY_STYLES[category.id];
          return (
            <div key={category.id} className="mb-3">
              <p className="px-2 py-1.5 text-[10px] font-semibold text-cds-text-secondary uppercase tracking-widest">
                {category.label}
              </p>
              <div className="space-y-0.5">
                {category.nodes.map((node) => {
                  const Icon = getIcon(node.icon);
                  return (
                    <div
                      key={node.type}
                      draggable
                      onDragStart={(e) => handleDragStart(e, node)}
                      className={cn(
                        'flex items-center gap-2.5 px-2 py-2 rounded-md cursor-grab',
                        'hover:bg-cds-layer-hover-02 active:cursor-grabbing',
                        'transition-colors duration-75'
                      )}
                      data-testid={`palette-node-${node.type}`}
                      role="button"
                      aria-label={`Drag ${node.label} node`}
                    >
                      <div
                        className={cn(
                          'w-7 h-7 rounded flex items-center justify-center shrink-0',
                          catStyles.iconBg
                        )}
                      >
```
