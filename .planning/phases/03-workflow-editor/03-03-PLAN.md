# Plan 03-03: Properties Panel, Top Bar, Bottom Bar, and Integration

<objective>
Complete the workflow editor by building the node properties panel (right panel), the editor top bar (pipeline name, version badge, Editor/Runs/Eval tabs, Save/Publish), and the editor bottom bar (Run Workflow, Chat, node/connection count). Wire all panels to the workflow store, implement save/publish/run actions via the API hooks, and add keyboard shortcuts (Delete node, Cmd+S). After this wave, the full workflow editor matches the Figma design and is functionally complete.
</objective>

<must_haves>
- Selecting a node opens the right panel with editable properties
- Properties panel has sections: Tool Version, Input Parameters, Resource Limits, Retry Policy, Metadata
- Top bar shows pipeline name, version badge, Editor/Runs/Eval tabs, Save and Publish buttons
- Bottom bar shows Run Workflow button, Chat button, node/connection count
- Save button calls saveWorkflowGraph API and clears dirty flag
- Publish button calls publishWorkflow API
- Run Workflow button calls runWorkflow API
- Cmd+S keyboard shortcut triggers save
- Delete/Backspace removes selected node
</must_haves>

```tsx
import { useCallback } from 'react';
import { useWorkflowStore } from '@store/workflowStore';
import { cn } from '@utils/cn';
import type { WorkflowNodeData } from '@/types/workflow';
import { NODE_CATEGORY_STYLES } from './nodes/nodeStyles';

/** Collapsible section wrapper */
function Section({ title, children, defaultOpen = true }: {
  title: string;
  children: React.ReactNode;
  defaultOpen?: boolean;
}) {
  return (
    <details open={defaultOpen} className="group border-b border-cds-border-subtle">
      <summary className="flex items-center justify-between px-3 py-2.5 cursor-pointer hover:bg-cds-background-hover">
        <span className="text-xs font-medium text-cds-text-secondary uppercase tracking-wider">
          {title}
        </span>
        <span className="text-cds-icon-secondary text-xs group-open:rotate-180 transition-transform">
          &#9660;
        </span>
      </summary>
      <div className="px-3 pb-3">{children}</div>
    </details>
  );
}

/** Key-value field row */
function FieldRow({ label, value, onChange, type = 'text' }: {
  label: string;
  value: string | number | undefined;
  onChange?: (val: string) => void;
  type?: string;
}) {
  return (
    <div className="flex flex-col gap-1 mb-2">
      <label className="text-[10px] text-cds-text-secondary uppercase tracking-wide">{label}</label>
      {onChange ? (
        <input
          type={type}
          value={value ?? ''}
          onChange={(e) => onChange(e.target.value)}
          className={cn(
            'w-full h-8 px-2.5 text-xs bg-cds-field-01 text-cds-text-primary',
            'border-0 border-b border-cds-border-strong',
            'focus:outline-2 focus:outline-cds-focus focus:outline-offset-[-2px]'
          )}
        />
      ) : (
        <span className="text-xs text-cds-text-primary py-1">{value ?? '—'}</span>
      )}
    </div>
  );
}

export default function NodePropertiesPanel() {
  const selectedNodeId = useWorkflowStore((s) => s.selectedNodeId);
  const nodes = useWorkflowStore((s) => s.nodes);
  const updateNodeData = useWorkflowStore((s) => s.updateNodeData);

  const selectedNode = nodes.find((n) => n.id === selectedNodeId);
  const nodeData = selectedNode?.data as WorkflowNodeData | undefined;

  const handleUpdateField = useCallback(
    (field: keyof WorkflowNodeData, value: unknown) => {
      if (!selectedNodeId) return;
      updateNodeData(selectedNodeId, { [field]: value } as Partial<WorkflowNodeData>);
    },
    [selectedNodeId, updateNodeData]
  );

  const handleUpdateResourceLimit = useCallback(
    (key: string, value: string) => {
      if (!selectedNodeId || !nodeData) return;
      updateNodeData(selectedNodeId, {
        resourceLimits: { ...nodeData.resourceLimits, [key]: value },
      });
    },
    [selectedNodeId, nodeData, updateNodeData]
  );

  const handleUpdateRetryPolicy = useCallback(
    (key: string, value: string) => {
      if (!selectedNodeId || !nodeData) return;
      const numValue = parseInt(value, 10);
      updateNodeData(selectedNodeId, {
        retryPolicy: {
          maxRetries: nodeData.retryPolicy?.maxRetries ?? 0,
          backoffMs: nodeData.retryPolicy?.backoffMs ?? 1000,
          [key]: isNaN(numValue) ? 0 : numValue,
        },
      });
    },
    [selectedNodeId, nodeData, updateNodeData]
  );

  if (!selectedNode || !nodeData) {
    return (
      <div className="flex flex-col items-center justify-center h-full text-center px-4" data-testid="no-node-selected">
        <p className="text-sm text-cds-text-secondary mb-1">No node selected</p>
        <p className="text-xs text-cds-text-placeholder">Click a node on the canvas to view its properties</p>
      </div>
    );
  }

  const catStyles = NODE_CATEGORY_STYLES[nodeData.category];

  return (
    <div className="flex flex-col h-full" data-testid="node-properties-panel">
      {/* Selection header */}
      <div className="px-3 py-3 border-b border-cds-border-subtle">
        <div className="flex items-center gap-2.5">
          <div className={cn('w-8 h-8 rounded-md flex items-center justify-center', catStyles.iconBg)}>
            <span className={cn('text-xs font-bold', catStyles.iconColor)}>
              {nodeData.label.charAt(0)}
            </span>
          </div>
          <div>
            <input
              value={nodeData.label}
              onChange={(e) => handleUpdateField('label', e.target.value)}
              className="text-sm font-medium text-cds-text-primary bg-transparent border-0 p-0 w-full focus:outline-none focus:ring-1 focus:ring-cds-focus rounded px-1 -ml-1"
              aria-label="Node name"
            />
            <p className="text-[10px] text-cds-text-secondary uppercase tracking-wide">
              {catStyles.label} &middot; {nodeData.nodeType}
            </p>
          </div>
        </div>
      </div>

      {/* Scrollable sections */}
      <div className="flex-1 overflow-y-auto">
        {/* Tool Version */}
        <Section title="Tool Version">
          <FieldRow
            label="Version"
            value={nodeData.toolVersion}
            onChange={(val) => handleUpdateField('toolVersion', val)}
          />
        </Section>

        {/* Input Parameters */}
        <Section title="Input Parameters">
          {nodeData.inputParameters ? (
            Object.entries(nodeData.inputParameters).map(([key, val]) => (
              <FieldRow
                key={key}
                label={key}
                value={typeof val === 'object' ? JSON.stringify(val) : String(val ?? '')}
                onChange={(newVal) => {
                  if (!selectedNodeId) return;
                  updateNodeData(selectedNodeId, {
                    inputParameters: {
                      ...nodeData.inputParameters,
                      [key]: newVal,
                    },
                  });
                }}
              />
            ))
          ) : (
            <p className="text-xs text-cds-text-placeholder py-2">No input parameters configured</p>
          )}
        </Section>

        {/* Resource Limits */}
        <Section title="Resource Limits">
          <FieldRow
            label="CPU"
            value={nodeData.resourceLimits?.cpu}
            onChange={(val) => handleUpdateResourceLimit('cpu', val)}
          />
          <FieldRow
            label="Memory"
            value={nodeData.resourceLimits?.memory}
            onChange={(val) => handleUpdateResourceLimit('memory', val)}
          />
          <FieldRow
            label="Timeout (s)"
            value={nodeData.resourceLimits?.timeout}
            onChange={(val) => handleUpdateResourceLimit('timeout', val)}
            type="number"
          />
        </Section>

        {/* Retry Policy */}
        <Section title="Retry Policy">
          <FieldRow
            label="Max Retries"
            value={nodeData.retryPolicy?.maxRetries}
            onChange={(val) => handleUpdateRetryPolicy('maxRetries', val)}
            type="number"
          />
          <FieldRow
            label="Backoff (ms)"
            value={nodeData.retryPolicy?.backoffMs}
            onChange={(val) => handleUpdateRetryPolicy('backoffMs', val)}
            type="number"
          />
        </Section>

        {/* Metadata (read-only) */}
        <Section title="Metadata" defaultOpen={false}>
          <FieldRow label="Created" value={nodeData.metadata?.createdAt ? new Date(nodeData.metadata.createdAt).toLocaleDateString() : undefined} />
          <FieldRow label="Last Run" value={nodeData.metadata?.lastRunAt ? new Date(nodeData.metadata.lastRunAt).toLocaleDateString() : undefined} />
          <FieldRow label="Avg Cost" value={nodeData.metadata?.avgCostUsd != null ? `$${nodeData.metadata.avgCostUsd.toFixed(2)}` : undefined} />
        </Section>
      </div>
    </div>
  );
}
```

```tsx
import { useCallback } from 'react';
import { useNavigate } from 'react-router-dom';
import { ArrowLeft, Save, Upload } from 'lucide-react';
import { useWorkflowStore } from '@store/workflowStore';
import { useSaveWorkflow, usePublishWorkflow } from '@hooks/useWorkflow';
import Button from '@components/ui/Button';
import Badge from '@components/ui/Badge';
import Tabs from '@components/ui/Tabs';
import { cn } from '@utils/cn';

const EDITOR_TABS = [
  { id: 'editor', label: 'Editor' },
  { id: 'runs', label: 'Runs' },
  { id: 'eval', label: 'Eval' },
];

export default function WorkflowEditorTopBar() {
  const navigate = useNavigate();
  const metadata = useWorkflowStore((s) => s.metadata);
  const nodes = useWorkflowStore((s) => s.nodes);
  const edges = useWorkflowStore((s) => s.edges);
  const isDirty = useWorkflowStore((s) => s.isDirty);
  const editorTab = useWorkflowStore((s) => s.editorTab);
  const setEditorTab = useWorkflowStore((s) => s.setEditorTab);
  const markClean = useWorkflowStore((s) => s.markClean);
  const setVersionStatus = useWorkflowStore((s) => s.setVersionStatus);

  const saveMutation = useSaveWorkflow();
  const publishMutation = usePublishWorkflow();

  const handleSave = useCallback(async () => {
    if (!metadata) return;
    await saveMutation.mutateAsync({
      workflowId: metadata.id,
      request: { nodes, edges, name: metadata.name, description: metadata.description },
    });
    markClean();
  }, [metadata, nodes, edges, saveMutation, markClean]);

  const handlePublish = useCallback(async () => {
    if (!metadata) return;
    // Save first if dirty
    if (isDirty) {
      await saveMutation.mutateAsync({
        workflowId: metadata.id,
        request: { nodes, edges, name: metadata.name, description: metadata.description },
      });
      markClean();
    }
    await publishMutation.mutateAsync({ workflowId: metadata.id });
    setVersionStatus('published');
  }, [metadata, nodes, edges, isDirty, saveMutation, publishMutation, markClean, setVersionStatus]);

  const versionBadgeVariant = metadata?.versionStatus === 'published' ? 'success' : 'default';
  const versionLabel = metadata
    ? `v${metadata.version} ${metadata.versionStatus}`
    : '';

  return (
    <div
      className="h-12 flex items-center justify-between px-4 border-b border-cds-border-subtle bg-white shrink-0"
      data-testid="workflow-editor-top-bar"
    >
      {/* Left: Back + Pipeline name + Version badge */}
      <div className="flex items-center gap-3 min-w-0">
        <button
          onClick={() => navigate('/workflows')}
          className="w-8 h-8 flex items-center justify-center text-cds-icon-secondary hover:text-cds-text-primary rounded-md hover:bg-cds-background-hover transition-colors"
          aria-label="Back to workflows"
        >
          <ArrowLeft size={16} />
        </button>
        <div className="flex items-center gap-2 min-w-0">
          <h1 className="text-sm font-medium text-cds-text-primary truncate max-w-[240px]">
            {metadata?.name ?? 'Untitled Workflow'}
          </h1>
          {metadata && (
            <Badge variant={versionBadgeVariant} className="shrink-0">
              {versionLabel}
            </Badge>
          )}
          {isDirty && (
            <span className="w-2 h-2 rounded-full bg-amber-400 shrink-0" title="Unsaved changes" />
          )}
        </div>
      </div>

      {/* Center: Editor/Runs/Eval tabs */}
      <div className="absolute left-1/2 -translate-x-1/2">
        <Tabs
          tabs={EDITOR_TABS}
          activeTab={editorTab}
          onChange={(tab) => setEditorTab(tab as 'editor' | 'runs' | 'eval')}
        />
      </div>

      {/* Right: Save + Publish */}
      <div className="flex items-center gap-2">
        <Button
          variant="secondary"
          size="sm"
          icon={<Save size={14} />}
          onClick={handleSave}
          disabled={!isDirty || saveMutation.isPending}
          aria-label="Save workflow"
        >
          {saveMutation.isPending ? 'Saving...' : 'Save'}
        </Button>
        <Button
          variant="primary"
          size="sm"
          icon={<Upload size={14} />}
          onClick={handlePublish}
          disabled={publishMutation.isPending}
          aria-label="Publish workflow"
        >
          {publishMutation.isPending ? 'Publishing...' : 'Publish'}
        </Button>
      </div>
    </div>
  );
}
```
