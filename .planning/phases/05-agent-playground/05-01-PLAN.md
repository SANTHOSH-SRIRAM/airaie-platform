# Plan 05-01: Types, Store, API, Session List, Chat Interface, Top Bar, and Page

<objective>
Define all agent playground types, create a Zustand store for session and chat state, build the API layer with mock data, create React Query hooks, build the SessionList sidebar component, the ChatInterface with inline ToolCallProposal cards, the AgentPlaygroundTopBar, and the AgentPlaygroundPage shell. After this wave, users can view a session list, switch sessions, see a chat conversation with tool call proposals, and interact with the top bar tabs and session controls.
</objective>

<must_haves>
- AgentSession, ChatMessage, ToolCallProposal, DecisionTraceEntry, AgentMetrics types exist
- agentPlaygroundStore manages activeSessionId, sessions list, messages map, sending state
- API module has session CRUD, message send, and mock data fallback
- React Query hooks wrap all agent playground API calls
- SessionList renders sessions with search, status badges, and new-session button
- ChatInterface renders user messages (right-aligned) and agent messages (left-aligned with avatar)
- ToolCallProposalCard renders inline with confidence, reasoning, inputs, cost, alternatives
- AgentPlaygroundTopBar renders agent name, version badge, tabs, session selector
- AgentPlaygroundPage composes all sub-components in a page layout
</must_haves>

```ts
import { create } from 'zustand';
import type { AgentSession, ChatMessage, AgentMetrics, PolicyStatus, DecisionTraceEntry, AgentPlaygroundStats } from '@/types/agentPlayground';

interface AgentPlaygroundStore {
  // State
  sessions: AgentSession[];
  activeSessionId: string | null;
  messages: Record<string, ChatMessage[]>; // keyed by sessionId
  decisionTrace: Record<string, DecisionTraceEntry[]>; // keyed by sessionId
  metrics: Record<string, AgentMetrics>; // keyed by sessionId
  policyStatus: PolicyStatus;
  stats: AgentPlaygroundStats;
  isSending: boolean;
  isAgentRunning: boolean;
  searchQuery: string;

  // Actions
  setSessions: (sessions: AgentSession[]) => void;
  setActiveSessionId: (id: string | null) => void;
  setMessages: (sessionId: string, messages: ChatMessage[]) => void;
  addMessage: (sessionId: string, message: ChatMessage) => void;
  setDecisionTrace: (sessionId: string, entries: DecisionTraceEntry[]) => void;
  setMetrics: (sessionId: string, metrics: AgentMetrics) => void;
  setPolicyStatus: (status: PolicyStatus) => void;
  setStats: (stats: AgentPlaygroundStats) => void;
  setIsSending: (sending: boolean) => void;
  setIsAgentRunning: (running: boolean) => void;
  setSearchQuery: (query: string) => void;
  clearSession: (sessionId: string) => void;
}

export const useAgentPlaygroundStore = create<AgentPlaygroundStore>((set) => ({
  sessions: [],
  activeSessionId: null,
  messages: {},
  decisionTrace: {},
  metrics: {},
  policyStatus: { autoApproveThreshold: 0.9, autoApproveEnabled: false },
  stats: { totalSessions: 0, avgConfidence: 0, successRate: 0 },
  isSending: false,
  isAgentRunning: false,
  searchQuery: '',

  setSessions: (sessions) => set({ sessions }),
  setActiveSessionId: (id) => set({ activeSessionId: id }),
  setMessages: (sessionId, messages) =>
    set((s) => ({ messages: { ...s.messages, [sessionId]: messages } })),
  addMessage: (sessionId, message) =>
    set((s) => ({
      messages: {
        ...s.messages,
        [sessionId]: [...(s.messages[sessionId] ?? []), message],
      },
    })),
  setDecisionTrace: (sessionId, entries) =>
    set((s) => ({ decisionTrace: { ...s.decisionTrace, [sessionId]: entries } })),
  setMetrics: (sessionId, metrics) =>
    set((s) => ({ metrics: { ...s.metrics, [sessionId]: metrics } })),
  setPolicyStatus: (status) => set({ policyStatus: status }),
  setStats: (stats) => set({ stats }),
  setIsSending: (sending) => set({ isSending: sending }),
  setIsAgentRunning: (running) => set({ isAgentRunning: running }),
  setSearchQuery: (query) => set({ searchQuery: query }),
  clearSession: (sessionId) =>
    set((s) => ({
      messages: { ...s.messages, [sessionId]: [] },
      decisionTrace: { ...s.decisionTrace, [sessionId]: [] },
      metrics: {
        ...s.metrics,
        [sessionId]: { iterations: 0, totalCost: 0, budgetRemaining: 100, durationMs: 0, timeoutMs: 300000 },
      },
    })),
}));
```
