# Plan 06-01: Types, API, Hooks, Filter Sidebar, and Tool Cards

<objective>
Define the full Tool Registry type system, create an API layer with realistic mock data (14 tools spanning all categories), build a Zustand store for filter/selection state, create React Query hooks, and implement the FilterSidebar, ToolCard, and ToolCardGrid components. This wave delivers the browsing and filtering experience so Wave 2 can add the properties panel, action bar, and routing integration.
</objective>

<must_haves>
- Tool, ToolVersion, ToolContract, ToolFilter types are defined
- API module returns 14 mock tools with realistic engineering tool data
- toolRegistryStore manages filters (status, category, adapter), sort, selected tool ID, and search query
- React Query hook wraps tool listing with filter support
- FilterSidebar renders checkbox groups for status, category, adapter with counts and "Clear all"
- ToolCard renders icon, name, description, version badge, adapter/category tags, usage stats, status badge
- ToolCardGrid renders filtered tools in a responsive 2-column card grid
- ToolRegistryPage composes ToolCardGrid as the main content area
</must_haves>

export type ToolStatus = 'published' | 'draft' | 'deprecated';

export interface ToolContractField {
  name: string;
  type: string; // e.g. 'file:step', 'number', 'string', 'json', 'file:vtk'
  description?: string;
  required?: boolean;
  default?: unknown;
}

export interface ToolContract {
  inputs: ToolContractField[];
  outputs: ToolContractField[];
}

// --- Execution Config ---
export interface ToolExecutionConfig {
  adapter: ToolAdapter;
  image?: string;       // e.g. 'airaie/fea-solver:2.1.0'
  registry?: string;    // e.g. 'ghcr.io/airaie'
  entrypoint?: string;
}

// --- Resource Limits ---
export interface ToolLimits {
  cpu: string;      // e.g. '4 cores'
  memory: string;   // e.g. '8 GB'
  timeout: string;  // e.g. '30 min'
}

// --- Sandbox Policy ---
export interface ToolSandboxPolicy {
  network: 'deny' | 'allow';
  filesystem?: 'read-only' | 'read-write';
  gpu?: boolean;
}

// --- Tool Version ---
export interface ToolVersion {
  version: string;      // e.g. 'v2.1.0'
  status: ToolStatus;
  createdAt: string;    // ISO date
  changelog?: string;
}

// --- Tool ---
export interface Tool {
  id: string;
  name: string;
  description: string;
  icon: string;          // lucide-react icon name
  status: ToolStatus;
  category: ToolCategory;
  adapter: ToolAdapter;
  currentVersion: string;

  execution: ToolExecutionConfig;
  limits: ToolLimits;

  versionCount: number;

  usageCount: number;
  createdAt: string;

}

// --- Filter State ---
export interface ToolFilter {
  status: ToolStatus[];
  category: ToolCategory[];
  adapter: ToolAdapter[];
  search: string;
  sortBy: 'name' | 'usage' | 'updated' | 'cost';

```ts
import { create } from 'zustand';
import type { ToolStatus, ToolCategory, ToolAdapter, ToolFilter } from '@/types/tool';

export interface ToolRegistryState {
  // Filter state
  filters: ToolFilter;
  // Selection
  selectedToolId: string | null;

  // Filter actions
  toggleStatusFilter: (status: ToolStatus) => void;
  toggleCategoryFilter: (category: ToolCategory) => void;
  toggleAdapterFilter: (adapter: ToolAdapter) => void;
  setSearchQuery: (query: string) => void;
  setSortBy: (sortBy: ToolFilter['sortBy']) => void;
  clearFilters: () => void;

  // Selection actions
  selectTool: (toolId: string | null) => void;
}

const DEFAULT_FILTERS: ToolFilter = {
  status: [],
  category: [],
  adapter: [],
  search: '',
  sortBy: 'name',
};

export const useToolRegistryStore = create<ToolRegistryState>((set) => ({
  filters: { ...DEFAULT_FILTERS },
  selectedToolId: null,

  toggleStatusFilter: (status) =>
    set((s) => ({
      filters: {
        ...s.filters,
        status: s.filters.status.includes(status)
          ? s.filters.status.filter((v) => v !== status)
          : [...s.filters.status, status],
      },
    })),

  toggleCategoryFilter: (category) =>
    set((s) => ({
      filters: {
        ...s.filters,
        category: s.filters.category.includes(category)
          ? s.filters.category.filter((v) => v !== category)
          : [...s.filters.category, category],
      },
    })),

  toggleAdapterFilter: (adapter) =>
    set((s) => ({
      filters: {
        ...s.filters,
        adapter: s.filters.adapter.includes(adapter)
          ? s.filters.adapter.filter((v) => v !== adapter)
          : [...s.filters.adapter, adapter],
      },
    })),

  setSearchQuery: (search) =>
    set((s) => ({ filters: { ...s.filters, search } })),

  setSortBy: (sortBy) =>
    set((s) => ({ filters: { ...s.filters, sortBy } })),

  clearFilters: () =>
    set({ filters: { ...DEFAULT_FILTERS } }),

  selectTool: (toolId) =>
    set({ selectedToolId: toolId }),
}));
```

```ts
import { useQuery } from '@tanstack/react-query';
import { useMemo } from 'react';
import { fetchTools, fetchTool } from '@api/tools';
import { useToolRegistryStore } from '@store/toolRegistryStore';
import type { Tool, ToolFilter } from '@/types/tool';

/** Apply filters + search + sort to tool list */
function applyFilters(tools: Tool[], filters: ToolFilter): Tool[] {
  let result = tools;

  // Status filter
  if (filters.status.length > 0) {
    result = result.filter((t) => filters.status.includes(t.status));
  }
  // Category filter
  if (filters.category.length > 0) {
    result = result.filter((t) => filters.category.includes(t.category));
  }
  // Adapter filter
  if (filters.adapter.length > 0) {
    result = result.filter((t) => filters.adapter.includes(t.adapter));
  }
  // Search
  if (filters.search) {
    const q = filters.search.toLowerCase();
    result = result.filter(
      (t) =>
        t.name.toLowerCase().includes(q) ||
        t.description.toLowerCase().includes(q)
    );
  }
  // Sort
  result = [...result].sort((a, b) => {
    switch (filters.sortBy) {
      case 'usage': return b.usageCount - a.usageCount;
      case 'updated': return new Date(b.updatedAt).getTime() - new Date(a.updatedAt).getTime();
      case 'cost': return parseFloat(a.costPerRun.replace(/[^0-9.]/g, '')) - parseFloat(b.costPerRun.replace(/[^0-9.]/g, ''));
      default: return a.name.localeCompare(b.name);
    }
  });

  return result;
}

/** Fetch all tools with client-side filtering */
export function useTools() {
  const filters = useToolRegistryStore((s) => s.filters);
  const query = useQuery({
    queryKey: ['tools'],
    queryFn: fetchTools,
    staleTime: 5 * 60 * 1000, // 5 min
  });

  const filteredTools = useMemo(
    () => (query.data ? applyFilters(query.data, filters) : []),
    [query.data, filters]
  );

  return { ...query, tools: filteredTools, allTools: query.data ?? [] };
}

/** Fetch a single tool by ID */
export function useTool(toolId: string | null) {
  return useQuery({
    queryKey: ['tool', toolId],
    queryFn: () => fetchTool(toolId!),
    enabled: !!toolId,
    staleTime: 5 * 60 * 1000,
  });
}
```
