```ts
import type { Card } from '@/types/card';
import type { Board } from '@/types/board';

export interface CardModeRules {
  mode: 'explore' | 'study' | 'release';
  // Editing
  canEditIntentGoalAndKpis: boolean;
  canPinInputs: boolean;
  canChangePipeline: boolean;
  canUseAgentInstead: boolean;
  // Evidence
  canAddManualEvidence: boolean;
  // Gates
  requiresPeerReview: boolean;
  requiresQaApproval: boolean;
  // Run
  canRun: boolean;
  // Promotion CTA on top bar
  promoteTarget: 'study' | 'release' | null;
}

export function useCardModeRules(card: Card | undefined, board: Board | undefined): CardModeRules {
  const mode = (board?.mode ?? 'explore') as CardModeRules['mode'];
  const hasFirstRun = Boolean(card?.started_at);

  return {
    mode,
    canEditIntentGoalAndKpis: mode === 'explore' || (mode === 'study' && !hasFirstRun),
    canPinInputs: mode === 'explore',
    canChangePipeline: mode === 'explore',
    canUseAgentInstead: mode === 'explore',
    canAddManualEvidence: mode !== 'release',
    requiresPeerReview: mode !== 'explore',
    requiresQaApproval: mode === 'release',
    canRun: Boolean(card?.intent_spec_id),
    promoteTarget: mode === 'explore' ? 'study' : mode === 'study' ? 'release' : null,
  };
}
```

```ts
import { api, apiClient } from './client';
import { unwrapList } from '@/utils/apiEnvelope';
import type { Gate } from '@/types/gate';

export async function listCardGates(cardId: string): Promise<Gate[]> {
  const res = await api<unknown>(`/v0/cards/${cardId}/gates`, { method: 'GET' });
  return unwrapList<Gate>(res, 'gates');
}

export function approveGate(gateId: string, rationale?: string): Promise<void> {
  return apiClient.post(`/v0/gates/${gateId}/approve`, { rationale });
}

export function rejectGate(gateId: string, rationale: string): Promise<void> {
  return apiClient.post(`/v0/gates/${gateId}/reject`, { rationale });
}

export function waiveGate(gateId: string, rationale: string): Promise<void> {
  return apiClient.post(`/v0/gates/${gateId}/waive`, { rationale });
}
```

Create `frontend/src/hooks/useGates.ts`:

```ts
import { useQuery, useMutation, useQueryClient } from '@tanstack/react-query';
import { listCardGates, approveGate, rejectGate, waiveGate } from '@api/gates';

export const gateKeys = {
  all: ['gates'] as const,
  forCard: (cardId: string) => ['gates', 'card', cardId] as const,
};

export function useGates(cardId: string | undefined) {
  return useQuery({
    queryKey: cardId ? gateKeys.forCard(cardId) : gateKeys.all,
    queryFn: () => listCardGates(cardId!),
    enabled: !!cardId,
    staleTime: 15_000,
  });
}

export function useApproveGate(cardId: string) {
  const qc = useQueryClient();
  return useMutation({
    mutationFn: ({ gateId, rationale }: { gateId: string; rationale?: string }) => approveGate(gateId, rationale),
    onSuccess: () => qc.invalidateQueries({ queryKey: gateKeys.forCard(cardId) }),
  });
}

// useRejectGate, useWaiveGate — same pattern, with required rationale
```

Add a `Gate` type at `frontend/src/types/gate.ts` if missing.
