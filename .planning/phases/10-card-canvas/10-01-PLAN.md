# Plan 10-01 — Editor Framework + Persistence

<objective>
Stand up the Tiptap-based editor framework that will host Phase 10's typed governance blocks (delivered in 10-02 onward). Specifically: install Tiptap directly (no BlockNote), define the AirAIE block schema as a TypeScript discriminated union, build the `useAirAirEditor` hook + `<AirAirEditor>` component, persist edits via a new `cards.body_blocks` JSONB column with optimistic concurrency, auto-migrate existing Cards into a default block tree on first canvas open, and mount a feature-flagged route `/cards/:id?canvas=1` rendering an empty canvas (with placeholder cards for typed blocks pending in 10-02).

This wave intentionally ships **NO typed-governance NodeViews**. The canvas opens, persists, migrates, and renders text/headings/lists/callouts/dividers — but `intentBlock`, `inputBlock`, `methodBlock`, etc. render as a generic `<TypedBlockPlaceholder>` saying "Rendered in Wave 2 (10-02)". This proves the framework end-to-end with minimum surface area.
</objective>

<must_haves>
- `@tiptap/core@^3.13.0`, `@tiptap/react@^3.13.0`, `@tiptap/pm@^3.13.0`, `@tiptap/starter-kit@^3.13.0`, `@tiptap/extension-placeholder@^3.13.0` installed
- `frontend/src/editor/schema.ts` defining the AirAIE block schema (10 typed-governance node types declared with attr shapes; layout nodes from StarterKit)
- `frontend/src/types/cardBlocks.ts` exporting TypeScript types: `CardBodyDoc`, `BlockNode`, discriminated-union per node `type`
- `useAirAirEditor(card, options)` hook that constructs a configured Tiptap editor with our schema + extensions
- `<AirAirEditor>` component that mounts `<EditorContent>` with our keyboard shortcuts and styling
- `<TypedBlockPlaceholder kind={...}>` — temporary NodeView rendering a labelled placeholder for typed blocks until 10-02 lands their real components
- New backend migration `031_card_body_blocks.sql` adding `body_blocks JSONB` and `body_blocks_version INTEGER DEFAULT 1`
- Backend handler `PATCH /v0/cards/:id/body` accepting `{ body_blocks, expected_version }`, returning new version or 409 on mismatch
- Frontend API + hook: `updateCardBody(cardId, body, expectedVersion)` and `useUpdateCardBody`
- `generateDefaultBody(card, intent, plan, latestRun, evidence, gates)` — pure function emitting a Tiptap doc from current entity state for first-open migration
- `CardCanvasPage` at route `/cards/:cardId` (with `?canvas=1` feature flag → renders canvas; otherwise renders Phase 8's `CardDetailPage` unchanged). Lazy-loaded.
- `CardTopBar` gains a small "Try canvas" link toggling `?canvas=1` (Phase 10a entry point for users)
- Debounced autosave (1500ms after last change) + explicit save on `beforeunload`
- Optimistic concurrency: 409 → re-fetch + show inline conflict toast (3-way merge deferred to 10e)
- Vitest covers: schema serialization round-trip, migration generates correct default doc per Card state, AirAirEditor mounts text + placeholder NodeViews, CardCanvasPage routes correctly with/without feature flag
- `tsc --noEmit` (default + strict) clean

```ts
// AirAIE block schema — Tiptap-compatible JSON shape
// Mirrors what `editor.getJSON()` produces but typed.

export type LayoutBlockType =
  | 'paragraph'
  | 'heading'        // attrs.level: 1 | 2 | 3
  | 'bulletList'
  | 'orderedList'
  | 'listItem'
  | 'blockquote'
  | 'codeBlock'
  | 'horizontalRule'
  | 'callout';       // attrs.variant: 'info' | 'warning' | 'success'

export type TypedBlockType =
  | 'intentBlock'
  | 'inputBlock'
  | 'kpiBlock'
  | 'methodBlock'
  | 'runBlock'
  | 'resultBlock'
  | 'evidenceBlock'
  | 'gateBlock'
  | 'embedCardBlock'
  | 'embedRecordBlock'
  | 'aiAssistBlock';

export type BlockType = LayoutBlockType | TypedBlockType;

export interface BaseBlockNode {
  type: BlockType;
  attrs?: Record<string, unknown>;
  content?: BlockNode[];
  marks?: { type: string; attrs?: Record<string, unknown> }[];
  text?: string;
}

// Discriminated unions for the typed blocks (validate attrs shape):
export interface IntentBlockNode extends BaseBlockNode {
  type: 'intentBlock';
  attrs: { intentSpecId: string };
}

export interface InputBlockNode extends BaseBlockNode {
  type: 'inputBlock';
  attrs: { artifactId: string; portName?: string };
}

export interface KpiBlockNode extends BaseBlockNode {
  type: 'kpiBlock';
  attrs: { metricKey: string; operator: 'between' | 'less_than' | 'greater_than' | 'eq'; threshold: number | [number, number] };
}

export interface MethodBlockNode extends BaseBlockNode {
  type: 'methodBlock';
  attrs: { planId: string };
}

export interface RunBlockNode extends BaseBlockNode {
  type: 'runBlock';
  attrs: Record<string, never>;
}

export interface ResultBlockNode extends BaseBlockNode {
  type: 'resultBlock';
  attrs: { artifactId: string };
}

export interface EvidenceBlockNode extends BaseBlockNode {
  type: 'evidenceBlock';
  attrs: { evidenceId: string };
}

export interface GateBlockNode extends BaseBlockNode {
  type: 'gateBlock';
  attrs: { gateId: string };
}

export interface EmbedCardBlockNode extends BaseBlockNode {
  type: 'embedCardBlock';
  attrs: { cardId: string };
}

export interface EmbedRecordBlockNode extends BaseBlockNode {
  type: 'embedRecordBlock';
  attrs: { recordId: string };
}

export interface AiAssistBlockNode extends BaseBlockNode {
  type: 'aiAssistBlock';
  attrs: { conversationId: string };
}

export type BlockNode =
  | BaseBlockNode  // for layout types
  | IntentBlockNode
  | InputBlockNode
  | KpiBlockNode
  | MethodBlockNode
  | RunBlockNode
  | ResultBlockNode
  | EvidenceBlockNode
  | GateBlockNode
  | EmbedCardBlockNode
  | EmbedRecordBlockNode
  | AiAssistBlockNode;

export interface CardBodyDoc {
  type: 'doc';
  content: BlockNode[];
  // Frontend-only; not persisted: editor uses this for client-state.
}

export const TYPED_BLOCK_KINDS: readonly TypedBlockType[] = [
  'intentBlock', 'inputBlock', 'kpiBlock', 'methodBlock', 'runBlock',
  'resultBlock', 'evidenceBlock', 'gateBlock',
  'embedCardBlock', 'embedRecordBlock', 'aiAssistBlock',
] as const;

export function isTypedBlock(b: BlockNode): b is BlockNode & { type: TypedBlockType } {
  return TYPED_BLOCK_KINDS.includes(b.type as TypedBlockType);
}
```

```ts
import { useEditor, type Editor } from '@tiptap/react';
import { useEffect, useRef } from 'react';
import { airAirExtensions } from './extensions';
import type { CardBodyDoc } from '@/types/cardBlocks';

export interface UseAirAirEditorOptions {
  initialDoc: CardBodyDoc | null;
  editable?: boolean;        // false in Release mode for read-only
  onChange?: (doc: CardBodyDoc) => void;
  onIdle?: (doc: CardBodyDoc) => void;  // fired 1500ms after last change (debounced save trigger)
}

export function useAirAirEditor(opts: UseAirAirEditorOptions): Editor | null {
  const onIdleTimer = useRef<ReturnType<typeof setTimeout> | null>(null);

  const editor = useEditor({
    extensions: airAirExtensions,
    content: opts.initialDoc ?? { type: 'doc', content: [{ type: 'paragraph' }] },
    editable: opts.editable !== false,
    onUpdate: ({ editor }) => {
      const doc = editor.getJSON() as unknown as CardBodyDoc;
      opts.onChange?.(doc);
      // Debounce idle save:
      if (onIdleTimer.current) clearTimeout(onIdleTimer.current);
      onIdleTimer.current = setTimeout(() => {
        opts.onIdle?.(doc);
      }, 1500);
    },
  }, [opts.initialDoc]);

  // beforeunload — flush pending save synchronously
  useEffect(() => {
    function flush() {
      if (onIdleTimer.current) {
        clearTimeout(onIdleTimer.current);
        if (editor) {
          opts.onIdle?.(editor.getJSON() as unknown as CardBodyDoc);
        }
      }
    }
    window.addEventListener('beforeunload', flush);
    return () => {
      flush();
      window.removeEventListener('beforeunload', flush);
    };
  }, [editor, opts.onIdle]);

  return editor;
}
```

```ts
import type { Card } from '@/types/card';
import type { IntentSpec } from '@/types/intent';
import type { Plan } from '@/types/plan';
import type { Run } from '@/types/run';
import type { CardEvidence } from '@/types/evidence';
import type { Gate } from '@/types/gate';
import type { CardBodyDoc, BlockNode } from '@/types/cardBlocks';

export function generateDefaultBody(args: {
  card: Card;
  intent: IntentSpec | null;
  plan: Plan | null;
  latestRun: Run | null;
  evidence: CardEvidence[];
  gates: Gate[];
}): CardBodyDoc {
  const { card, intent, plan, latestRun, evidence, gates } = args;
  const blocks: BlockNode[] = [];

  // Intent block (if present)
  if (intent) {
    blocks.push({ type: 'intentBlock', attrs: { intentSpecId: intent.id } });
  } else {
    // Empty placeholder paragraph inviting the user to start
    blocks.push({
      type: 'paragraph',
      content: [{ type: 'text', text: 'Type / to add an Intent, or click Edit Intent above.' }],
    });
  }

  // Inputs
  for (const input of intent?.inputs ?? []) {
    blocks.push({
      type: 'inputBlock',
      attrs: { artifactId: input.artifact_id, portName: input.port },
    });
  }

  // KPIs
  for (const kpi of intent?.acceptance ?? []) {
    blocks.push({
      type: 'kpiBlock',
      attrs: { metricKey: kpi.kpi, operator: kpi.operator, threshold: kpi.value },
    });
  }

  // Method
  if (plan) {
    blocks.push({ type: 'methodBlock', attrs: { planId: plan.id } });
  }

  // Run (always present once we have a method)
  if (plan) {
    blocks.push({ type: 'runBlock', attrs: {} });
  }

  // Results (if run completed)
  if (latestRun?.status === 'completed') {
    for (const out of latestRun.outputs ?? []) {
      blocks.push({ type: 'resultBlock', attrs: { artifactId: out.artifact_id } });
    }
  }

  // Evidence rows (if any)
  for (const e of evidence) {
    blocks.push({ type: 'evidenceBlock', attrs: { evidenceId: e.id } });
  }

  // Gates (if any required by Mode)
  for (const g of gates) {
    blocks.push({ type: 'gateBlock', attrs: { gateId: g.id } });
  }

  return { type: 'doc', content: blocks };
}
```
